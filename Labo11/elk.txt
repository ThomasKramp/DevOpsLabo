// https://www.elastic.co/guide/en/beats/filebeat/7.6/directory-layout.html

// Ubuntu
curl -L -O https://artifacts.elastic.co/downloads/beats/filebeat/filebeat-7.6.2-amd64.deb

sudo dpkg -i filebeat-7.6.2-amd64.deb

cd /etc/filebeat/
sudo nano filebeat.yml
	Edit: Change "enabled: false" --> "enabled: true"
	Edit: Un-comment "username: "elastic""
	Edit: Un-comment "password: "changeme""
	Add: "setup.kibana:
	        host: "localhost:5601""

sudo filebeat setup --index-management -E output.logstash.enabled=false -E 'output.elasticsearch.hosts=["localhost:9200"]'

sudo filebeat setup --dashboards

sudo filebeat setup -e \
  -E output.logstash.enabled=false \
  -E output.elasticsearch.hosts=['localhost:9200'] \
  -E output.elasticsearch.username=elastic \
  -E output.elasticsearch.password=changeme \
  -E setup.kibana.host=localhost:5601
  
sudo service filebeat start
